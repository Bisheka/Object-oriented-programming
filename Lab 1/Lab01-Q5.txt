public class MathOperations {

    // Method to add two integers
    public static int add(int a, int b) {
        return a + b;
    }

    // Method to multiply two integers
    public static int multiply(int a, int b) {
        return a * b;
    }

    // Method to calculate the square of an integer
    public static int square(int a) {
        return a * a;
    }

    public static void main(String[] args) {
        // Calculate (3 * 4 + 5 * 7)^2
        int expression1 = square(add(multiply(3, 4), multiply(5, 7)));

        // Calculate (4 + 7)^2 + (8 + 3)^2
        int expression2 = add(square(add(4, 7)), square(add(8, 3)));

        // Print results
        System.out.println("Result of expression 1: " + expression1);
        System.out.println("Result of expression 2: " + expression2);
    }
}
